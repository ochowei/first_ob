太好了，下面給你一支可直接執行的 **統一 Python Script**（用 `yfinance` 下載資料）來計算多檔 ETF 的「動能分數」、與 BIL 的基準比較，並依分數自動產生建議配置（可選擇對 BND 給 10% 保底配置）。

你可以把它存成 `momentum_etf.py`，然後用 `python momentum_etf.py` 執行；或用參數自訂標的、保底比例等。

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Momentum ETF Scorer & Allocator
- 依 Amber 姐姐公式計算動能分數：
  Score = 12*R1M + 4*R3M + 2*R6M + 1*R12M
- 與 BIL 比較，僅分數 > BIL 者列入配置池（其餘視為持有 BIL/現金）
- 可選擇對 BND 給最小配置（例如 10%）
- 預設標的：VOO, VTWO, VXUS, SCZ, SPYG, VNQ, MORT, BND, GLD, BIL

用法範例：
  python momentum_etf.py
  python momentum_etf.py --tickers VOO,SPYG,VXUS,BND,GLD,BIL --min-bnd 0.10 --hold BIL
"""

import argparse
from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple

import pandas as pd
import yfinance as yf


# -------- Config --------
DEFAULT_TICKERS = ["VOO", "VTWO", "VXUS", "SCZ", "SPYG", "VNQ", "MORT", "BND", "GLD", "BIL"]
BENCHMARK = "BIL"   # 與之比較的「無風險利率代表」
NEEDED_MONTHS = 13  # 至少需要跨 12 個月的資料點（+1月餘裕）


@dataclass
class Returns:
    r1m: float
    r3m: float
    r6m: float
    r12m: float


def nearest_price(series: pd.Series, target_ts: pd.Timestamp) -> float:
    """回傳最接近 target_ts 的收盤價。"""
    idx = series.index
    # 若剛好有精確日期就直接取；否則取距離最短的日期
    if target_ts in idx:
        return float(series.loc[target_ts])
    # 使用 asof（往前找最近），若沒有就退而求其次用最近日期
    try:
        val = series.asof(target_ts)  # type: ignore[attr-defined]
        if pd.isna(val):
            # asof 失敗時，改用距離最近的日期
            nearest = idx[(abs(idx - target_ts)).argmin()]
            return float(series.loc[nearest])
        return float(val)
    except Exception:
        nearest = idx[(abs(idx - target_ts)).argmin()]
        return float(series.loc[nearest])


def compute_period_returns(series: pd.Series, ref_date: Optional[pd.Timestamp] = None) -> Returns:
    """計算 1 / 3 / 6 / 12 個月區間報酬率。"""
    if ref_date is None:
        ref_date = series.index[-1]

    def ret(months: int) -> float:
        past_date = ref_date - pd.DateOffset(months=months)
        past_price = nearest_price(series, past_date)
        now_price = float(series.loc[ref_date])
        return (now_price / past_price) - 1.0

    return Returns(r1m=ret(1), r3m=ret(3), r6m=ret(6), r12m=ret(12))


def momentum_score(rets: Returns) -> float:
    """Amber 公式：12*1M + 4*3M + 2*6M + 1*12M。"""
    return 12 * rets.r1m + 4 * rets.r3m + 2 * rets.r6m + 1 * rets.r12m


def download_prices(tickers: List[str], period_years: int = 2) -> pd.DataFrame:
    """下載收盤價（日線）。抓 2 年以涵蓋 12 個月回溯並防止缺漏。"""
    df = yf.download(tickers, period=f"{period_years}y", interval="1d", auto_adjust=True, progress=False)["Close"]
    # 若單一標的時是 Series，轉成 DataFrame
    if isinstance(df, pd.Series):
        df = df.to_frame()
    # 去掉全為 NaN 的欄位
    df = df.dropna(how="all", axis=1)
    # 前向填補小缺口（假日差異等），但不跨標的彼此
    df = df.fillna(method="ffill")
    return df


def ensure_enough_history(series: pd.Series) -> bool:
    """檢查是否有足夠月份資料"""
    if series.dropna().shape[0] < 20:
        return False
    first = series.dropna().index[0]
    last = series.dropna().index[-1]
    months = (last.year - first.year) * 12 + (last.month - first.month)
    return months >= NEEDED_MONTHS


def allocate(weights_raw: Dict[str, float],
             min_bnd: float = 0.10,
             hold: str = "BIL") -> Tuple[Dict[str, float], Dict[str, float]]:
    """
    給出建議配置：
    - 僅將「分數 > BIL」的標的納入配置池，按分數比例分配
    - 對 BND 可選擇保底 min_bnd（若 BND 不在池中且在候選清單，將先撥出保底，再對剩餘資金按比例分配）
    - 其餘資金配置到 hold（'BIL' 或 'CASH'）
    回傳：(final_alloc, pool_weights) 其中 pool_weights 是未加保底前的原比例。
    """
    pool = {k: v for k, v in weights_raw.items() if v > 0}
    total = sum(pool.values())
    pool_weights = {k: (v / total) if total > 0 else 0.0 for k, v in pool.items()}

    final: Dict[str, float] = {}

    # 先處理 BND 保底
    reserved = 0.0
    if min_bnd > 0 and ("BND" in weights_raw):
        final["BND"] = max(min_bnd, pool_weights.get("BND", 0.0))
        reserved += final["BND"]

    # 剩餘按比例分配給其餘通過資產
    remaining = 1.0 - reserved
    if remaining < 0:
        remaining = 0.0

    for k, w in pool_weights.items():
        if k == "BND" and "BND" in final:
            continue
        final[k] = w * remaining

    # 若沒有通過資產或仍有未分配金額，補到 hold
    allocated = sum(final.values())
    if allocated < 1.0:
        final[hold] = final.get(hold, 0.0) + (1.0 - allocated)

    # 清理接近 0 的數
    final = {k: round(v, 4) for k, v in final.items() if v > 1e-6}
    return final, pool_weights


def main():
    parser = argparse.ArgumentParser(description="Momentum ETF Scorer & Allocator (Amber formula)")
    parser.add_argument("--tickers", type=str,
                        default=",".join(DEFAULT_TICKERS),
                        help="逗號分隔的代號清單（預設：Amber 文中的 ETF + BIL）")
    parser.add_argument("--min-bnd", type=float, default=0.10,
                        help="BND 最小配置比例（預設 0.10；設 0 代表不保底）")
    parser.add_argument("--hold", type=str, choices=["BIL", "CASH"], default="BIL",
                        help="未通過資產的資金去向（BIL 或 CASH；預設 BIL）")
    parser.add_argument("--years", type=int, default=2, help="下載幾年的日線資料（預設 2 年）")
    args = parser.parse_args()

    tickers = [t.strip().upper() for t in args.tickers.split(",") if t.strip()]
    if BENCHMARK not in tickers:
        tickers.append(BENCHMARK)  # 一定要有 BIL

    prices = download_prices(tickers, period_years=args.years)

    # 逐檔計算
    records = []
    ref_date = prices.index[-1]
    for t in tickers:
        if t not in prices.columns:
            continue
        series = prices[t].dropna()
        if series.empty or not ensure_enough_history(series):
            records.append({
                "Ticker": t,
                "RefDate": ref_date.date(),
                "R1M": None, "R3M": None, "R6M": None, "R12M": None,
                "Score": None
            })
            continue

        rets = compute_period_returns(series, ref_date)
        score = momentum_score(rets)
        records.append({
            "Ticker": t,
            "RefDate": ref_date.date(),
            "R1M": rets.r1m,
            "R3M": rets.r3m,
            "R6M": rets.r6m,
            "R12M": rets.r12m,
            "Score": score
        })

    df = pd.DataFrame.from_records(records)
    # 與 BIL 比較
    bil_score = float(df.loc[df["Ticker"] == BENCHMARK, "Score"].values[0]) if BENCHMARK in df["Ticker"].values else float("nan")
    df[">BIL"] = df["Score"] > bil_score

    # 排序：先通過者，再依分數由高到低
    df_sorted = df.sort_values(by=[">BIL", "Score"], ascending=[False, False]).reset_index(drop=True)

    # 組合建議（將分數 < BIL 視為不納入池；分數 <= 0 也不納入）
    pool_scores = {row.Ticker: float(row.Score) for _, row in df_sorted.iterrows()
                   if (row.Ticker != BENCHMARK) and pd.notna(row.Score) and (float(row.Score) > bil_score)}
    final_alloc, pool_weights = allocate(pool_scores, min_bnd=args.min_bnd, hold=args.hold)

    # 輸出
    pd.options.display.float_format = "{:,.4f}".format
    print("\n=== Momentum Scores (Amber formula) ===")
    print(df_sorted[["Ticker", "RefDate", "R1M", "R3M", "R6M", "R12M", "Score", ">BIL"]].to_string(index=False))

    print(f"\nBenchmark (BIL) Score: {bil_score:,.4f}")

    if len(pool_scores) == 0:
        print("\nNo assets beat BIL. 100% allocated to", args.hold)
        print({"BIL" if args.hold == "BIL" else "CASH": 1.0})
        return

    print("\n=== Passing Pool (raw proportional weights before min-BND) ===")
    for k, v in sorted(pool_weights.items(), key=lambda x: x[1], reverse=True):
        print(f"{k}: {v:,.4%}")

    print("\n=== Suggested Allocation (sum to 100%) ===")
    for k, v in sorted(final_alloc.items(), key=lambda x: x[1], reverse=True):
        print(f"{k}: {v:,.2%}")


if __name__ == "__main__":
    main()
```

### 使用說明

1. 安裝依賴：
    
    ```bash
    pip install yfinance pandas
    ```
    
2. 儲存為 `momentum_etf.py`。
    
3. 直接跑預設（含文中 ETF 與 BIL 基準）：
    
    ```bash
    python momentum_etf.py
    ```
    
4. 自訂標的／保底比例／持有資產：
    
    ```bash
    # 只評 VOO, SPYG, VXUS, BND, GLD + BIL；BND 保底 10%；未通過者資金配置到 BIL
    python momentum_etf.py --tickers VOO,SPYG,VXUS,BND,GLD,BIL --min-bnd 0.10 --hold BIL
    
    # 不要 BND 保底；未通過者留在現金
    python momentum_etf.py --min-bnd 0 --hold CASH
    ```
